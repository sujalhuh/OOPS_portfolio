import socket
import time


# ================= SENSOR INTERFACE CLASS =================
class SensorInterface:
    def __init__(self, sensor_type="default"):
        self._sensor_type = sensor_type
        self._data_buffer = []
        self._calibration_offset = 0.0
    
    def read_data(self):
        """Abstraction: Simple interface for complex sensing"""
        raw_data = self._acquire_raw_data()
        return self._process_data(raw_data)
    
    def _acquire_raw_data(self):
        """Encapsulation: Private method for data acquisition"""
        # Simulated sensor reading
        return {"value": 100, "timestamp": time.time()}
    
    def _process_data(self, raw_data):
        """Encapsulation: Private data processing"""
        return raw_data["value"] + self._calibration_offset



# ================= CONTROL CLIENT CLASS =================
class ControlClient:
    def __init__(self, host, port):
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sensor = SensorInterface()  # Composition
        
    def send_command(self, cmd):
        """Abstraction: Simple command interface"""
        self.socket.sendall((cmd + "\n").encode())
    
    def process_and_send(self):
        """Demonstrates OOP workflow"""
        data = self.sensor.read_data()
        
        if data > 150:
            self.send_command("ACTIVATE")
        else:
            self.send_command("STANDBY")

import socket


# ================= BASE ACTUATOR CLASS =================
class Actuator:
    def __init__(self, actuator_id):
        self._id = actuator_id
        self._state = "IDLE"
    
    def activate(self):
        """Polymorphism: Base method to be overridden"""
        self._state = "ACTIVE"
        print(f"Actuator {self._id} activated")
    
    def deactivate(self):
        self._state = "IDLE"



# ================= MOTOR ACTUATOR (INHERITANCE) =================
class MotorActuator(Actuator):
    def __init__(self, actuator_id, speed=0):
        super().__init__(actuator_id)  # Inheritance
        self._speed = speed
    
    def activate(self):
        """Polymorphism: Overridden method"""
        super().activate()
        self._set_speed(self._speed)
    
    def _set_speed(self, speed):
        """Encapsulation: Private speed control"""
        print(f"Motor {self._id} speed set to {speed}")



# ================= SYSTEM CONTROLLER =================
class SystemController:
    def __init__(self):
        # Dependency Injection through composition
        self.motor = MotorActuator("MOTOR_1", speed=75)
        self.server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        
    def run(self):
        """Main control loop"""
        while True:
            cmd = self._receive_command()
            self._process_command(cmd)
    
    def _process_command(self, cmd):
        """Encapsulation: Private command processing"""
        if cmd == "ACTIVATE":
            self.motor.activate()
        elif cmd == "STANDBY":
            self.motor.deactivate()
